find_package(PythonModules COMPONENTS sphinx REQUIRED)

endif()

if (MCRL2_ENABLE_DOC_PDFLATEX)
    find_package(LATEX COMPONENTS BIBTEX REQUIRED)
endif()

add_subdirectory(sphinx)
